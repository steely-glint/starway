Steps to create a BBB image:
1) 
put 
bone-debian-8.4-iot-armhf-2016-05-13-4gb.img
onto an SD card 
sudo dd bs=1m if=bone-debian-8.4-iot-armhf-2016-05-13-4gb.img of=/dev/rdisk1

2) insert the sd card in a 4gb BBB - power up with the boot button pressed
(near the large USB connector)

3) connect to the BBB by plugging a usb cable to the small USB 
connector and your mac.
Also plug in an ethernet connection to the internet - 
or use usbtethering on on an android phone

sudo cu -s 115200 -l /dev/cu.usbmodemFA1233 
(numbers may vary)

You'll need your mac password 

now login as debian with password of temppwd

4) Add thp user
sudo useradd -m -G admin thp 
sudo passwd thp
5) change the debian password
passwd 

6) replace the kernel
cd /opt/scripts/tools
./update_kernel.sh --bone-rt-kernel --lts-4_4

7)
reboot the BBB - holding down the boot button

login as thp

8)
sudo mkdir -p /project2/github/Yona-Appletree/
sudo chown -R thp /project2/github
cd /project2/github/Yona-Appletree/
git clone git://github.com/Yona-Appletree/LEDscape
cd LEDscape

9) install prustuff (follow README.md )
	sudo bash
        cp /boot/dtbs/$(uname -r)/am335x-boneblack.dtb{,.preledscape_bk}	
	cp am335x-boneblack.dtb /boot/dtbs/$(uname -r)/	
	cp am335x-bonegreen.dtb /boot/dtbs/$(uname -r)/	
	modprobe uio_pruss	
	vi /boot/uEnv.txt
10)
	sudo reboot
(holding down boot button)
log in as thp

cd /project2/github/Yona-Appletree/LEDscape
vi opc-server.c
change e131 port to 0 in
+       .e131_port = 0,
And
+       assert_int_range_inclusive("e131 UDP Port", 0, 65535, input_config->e131_port);

make

sudo ./install-service.sh

11)
clean up

sudo apt-get remove apache
sudo apt-get remove nodejs
sudo apt-get autoremove

12)
install java

scp thp@192.67.4.33:/Users/thp/Downloads/jdk-8u101-fcs-bin-b13-linux-arm-vfp-hflt-21_jun_2016.tar .

(or download a fresh version from oracle)
tar -xvf jdk-8u101-fcs-bin-b13-linux-arm-vfp-hflt-21_jun_2016.tar
rm jdk1.8.0_101/src.zip
sudo mv jdk1.8.0_101 /usr
sudo ln -s /usr/jdk1.8.0_101 /usr/java
echo 'JAVA_HOME=/usr/java' >> .profile
echo 'PATH=$PATH:$JAVA_HOME/bin' >> .profile
